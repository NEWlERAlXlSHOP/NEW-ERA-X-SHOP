    <script>
        // Get references to elements
        const menuBtn = document.getElementById('menu-btn');
        const mobileMenu = document.getElementById('mobile-menu');
        const accountsStore = document.getElementById('accounts-store');
        const ucStore = document.getElementById('uc-store');
        const switchPageBtn = document.getElementById('switch-page-btn');
        const switchPageBtnMobile = document.getElementById('switch-page-btn-mobile');
        const shopName = document.getElementById('shop-name');
        const shopTitle = document.getElementById('shop-title');
        const aboutSection = document.getElementById('about');
        const contactSection = document.getElementById('contact');

        // Initial state
        let isUcPage = false;
        
        // Function to switch between pages
        function switchPage() {
            if (isUcPage) {
                // Switch to Accounts Page
                accountsStore.classList.remove('hidden');
                ucStore.classList.add('hidden');
                switchPageBtn.textContent = 'NEW ERA X';
                switchPageBtnMobile.textContent = 'NEW ERA X';
                shopName.textContent = 'PUBG';
                shopName.classList.remove('text-pink-500');
                shopName.classList.add('text-cyan-400');
                shopTitle.textContent = 'АККАУНТЫ';
                aboutSection.classList.remove('bg-gray-900');
                aboutSection.classList.add('bg-gray-900');
                contactSection.classList.remove('bg-gray-950');
                contactSection.classList.add('bg-gray-950');
            } else {
                // Switch to UC Page
                accountsStore.classList.add('hidden');
                ucStore.classList.remove('hidden');
                switchPageBtn.textContent = 'МАГАЗИН АККАУНТОВ';
                switchPageBtnMobile.textContent = 'МАГАЗИН АККАУНТОВ';
                shopName.textContent = 'NEW ERA';
                shopName.classList.remove('text-cyan-400');
                shopName.classList.add('text-pink-500');
                shopTitle.textContent = 'X';
                aboutSection.classList.remove('bg-gray-900');
                aboutSection.classList.add('bg-gray-900');
                contactSection.classList.remove('bg-gray-950');
                contactSection.classList.add('bg-gray-950');
            }
            isUcPage = !isUcPage;
        }

        // Add event listeners for the page switch buttons
        switchPageBtn.addEventListener('click', switchPage);
        switchPageBtnMobile.addEventListener('click', switchPage);

        // Mobile menu toggle
        menuBtn.addEventListener('click', () => {
            mobileMenu.classList.toggle('hidden');
        });
        
        // Simple smooth scrolling for anchor links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                document.querySelector(this.getAttribute('href')).scrollIntoView({
                    behavior: 'smooth'
                });
            });
        });
    </script>
